import { db } from "./firebase";
import { collection, getDocs } from "firebase/firestore";

// Function to fetch brand data from Firestore
async function fetchBrands() {
  const brandsCollection = collection(db, "brands");
  const brandSnapshot = await getDocs(brandsCollection);
  const brandList = brandSnapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));

  console.log("Fetched Brands:", brandList); // Debugging: Check output in console
  return brandList;
}

export default fetchBrands;
